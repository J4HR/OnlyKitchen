    FOREIGN KEY (FriendUserID) REFERENCES Users(UserID)

    RecipeID INTEGER PRIMARY KEY AUTOINCREMENT,
    UserID INTEGER NOT NULL,

    Description TEXT,
    Instructions TEXT,
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);
